\def\installopt#1#2#3#4{%
  \installfont{MinionPro-Medium#3#1-#4oml}{%
     MnSymbolA#2                       encoding mn-symbol-a,%
     MnSymbolC#2                       encoding mn-symbol-c,%
     MnSymbolD#2                       encoding mn-symbol-d,%
     MnSymbolE#2                       encoding mn-symbol-e,%
     MinionPro-Medium#1-tosf-#4oml     encoding mn-oml-up,%
     MinionPro-MediumIt#1-tosf-#4oml   encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks,%
     tie%
  }{mn-oml-#3}{}{}{}{}{}%
}

\installfonts
