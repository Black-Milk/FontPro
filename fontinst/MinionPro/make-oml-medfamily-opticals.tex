\def\installopt#1#2#3#4{%
  \installfont{MinionPro-#3#1-#4oml}{%
     MnSymbolA#2                       encoding mn-symbol-a,%
     MnSymbolC#2                       encoding mn-symbol-c,%
     MnSymbolD#2                       encoding mn-symbol-d,%
     MnSymbolE#2                       encoding mn-symbol-e,%
     MinionPro-#1-tosf-#4oml           encoding mn-oml-up,%
     MinionPro-It#1-tosf-#4oml         encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks,%
     tie%
  }{mn-oml-#3}{}{}{}{}{}%
  \installfont{MinionPro-Semibold#3#1-#4oml}{%
     MnSymbolA-Bold#2                  encoding mn-symbol-a,%
     MnSymbolC-Bold#2                  encoding mn-symbol-c,%
     MnSymbolD-Bold#2                  encoding mn-symbol-d,%
     MnSymbolE-Bold#2                  encoding mn-symbol-e,%
     MinionPro-Semibold#1-tosf-#4oml   encoding mn-oml-up,%
     MinionPro-SemiboldIt#1-tosf-#4oml encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks,%
     tie%
  }{mn-oml-#3}{}{}{}{}{}%
  \installfont{MinionPro-Bold#3#1-#4oml}{%
     MnSymbolA-Bold#2                  encoding mn-symbol-a,%
     MnSymbolC-Bold#2                  encoding mn-symbol-c,%
     MnSymbolD-Bold#2                  encoding mn-symbol-d,%
     MnSymbolE-Bold#2                  encoding mn-symbol-e,%
     MinionPro-Bold#1-tosf-#4oml       encoding mn-oml-up,%
     MinionPro-BoldIt#1-tosf-#4oml     encoding mn-oml-it,%
     adjust-oml,%
     arrowhooks,%
     tie%
  }{mn-oml-#3}{}{}{}{}{}%
}
