\def\installsemiboldnonopt#1#2{%
  \installfont{#1}{%
     MnSymbolF-Bold10                  encoding mn-symbol-f-partialints,%
     #2                                encoding mn-symbol-f,%
     MyriadPro-SemiboldIt-integral     encoding mn-tint scaled 1140,%
     MyriadPro-SemiboldIt-integral-cn2 encoding mn-dint scaled 1140,%
     mn-intop-semibold-text,%
     mn-intop-semibold-disp-noopt%
  }{mn-symbol-fi}{}{}{}{}{}%
}

\installfonts
