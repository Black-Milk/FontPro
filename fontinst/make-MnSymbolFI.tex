\def\installopt#1#2{%
  \installfont{#1}{%
     MnSymbolF10                       encoding mn-symbol-f-partialints,%
     #2                                encoding mn-symbol-f,%
     MinionPro-It-integral             encoding mn-tint scaled 1100,%
     MinionPro-ItDisp-integral-cn1     encoding mn-dint scaled 1100,%
     mn-intop-text,%
     mn-intop-disp%
  }{mn-symbol-fi}{}{}{}{}{}%
}

\def\installsemiboldopt#1#2{%
  \installfont{#1}{%
     MnSymbolF-Bold10                      encoding mn-symbol-f-partialints,%
     #2                                    encoding mn-symbol-f,%
     MinionPro-SemiboldIt-integral         encoding mn-tint scaled 1140,%
     MinionPro-SemiboldItDisp-integral-cn1 encoding mn-dint scaled 1140,%
     mn-intop-semibold-text,%
     mn-intop-semibold-disp%
  }{mn-symbol-fi}{}{}{}{}{}%
}
